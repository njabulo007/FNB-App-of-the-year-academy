Python 3 Quick-Reference Sheet

Offline-ready syntax and structure overview

1. Variables & Types

Common Data Types

# Assignments are dynamic and can change type
x = 42                  # int
pi = 3.1415             # float
name = "Alice"          # str
active = True           # bool
empty = None            # null value
colors = ("red", "green") # tuple
unique = {"a", "b"}      # set

Type Checking & Conversion

# Inspect object type
print(type(x))              # <class 'int'>
print(isinstance(name, str)) # True

# Convert values
str_val = str(10)       # '10'
int_val = int("5")       # 5
float_val = float(3)    # 3.0

Common Built-ins

# Useful helpers
print(len(colors))         # 2
print(sorted([3, 1, 2])) # [1, 2, 3]

2. Control Flow

Conditional Statements

score = 85

if score >= 90:
    print("A")
elif score >= 80:
    print("B")
else:
    print("C")

# Ternary expression
level = "high" if score >= 80 else "low"

Loops

# For loop over items
fruits = ["apple", "banana"]
for fruit in fruits:
    print(fruit)

# Range with step
for i in range(1, 6, 2):
    print(i)             # 1, 3, 5

# While loop with break and continue
count = 0
while count < 5:
    count += 1
    if count == 3:
        continue
    print(count)
    if count == 4:
        break

3. Common Data Structures

Lists (Ordered, Mutable)

items = ["apple", "banana"]
items.append("cherry")  # Add item
items.insert(0, "orange") # Insert at index
first = items[0]         # Access element
subset = items[1:3]       # Slice access
items[1] = "blueberry" # Replace element

List Comprehension

evens = [n for n in range(10) if n % 2 == 0]

Dictionaries (Key-Value)

user = {
    "name": "Bob",
    "role": "Admin"
}
user["age"] = 25        # Add key/value
role = user.get("role")             # Safe access
for key, value in user.items():
    print(key, value)

Tuple & Set

point = (10, 20)   # Immutable tuple
letters = {"a", "b", "a"} # Set removes duplicates

4. Functions & Scope

Function Definition

def greet(name, greeting="Hello"):
    """Return a greeting for the given name."""
    return f"{greeting}, {name}!"

message = greet("Alice")

Arguments

# Positional, default, keyword
def add(a, b=0):
    return a + b

# Variable arguments
def show(*args, **kwargs):
    print(args)
    print(kwargs)

Scope

# Local and global scope
count = 0

def increment():
    global count
    count += 1

def outer():
    value = 1
    def inner():
        nonlocal value
        value += 1
    inner()

5. File I/O & Exceptions

Reading and Writing Files

# Safe handling with context manager
with open("log.txt", "w") as file:
    file.write("System initialized.\n")

with open("log.txt", "r") as file:
    content = file.read()

# Modes: 'r', 'w', 'a', 'x', 'b', 't'

Exception Handling

try:
    value = "12" + 3
except (TypeError, ValueError) as err:
    print("Invalid operation:", err)
else:
    print("Success" )
finally:
    print("Always runs.")

6. Object-Oriented Programming

Classes & Inheritance

class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "..."

    def __str__(self):
        return f"Animal({self.name})"

class Dog(Animal):
    def speak(self):
        return "Woof!"

    @classmethod
    def create_puppy(cls, name):
        return cls(name)

    @staticmethod
    def info():
        return "Dogs are mammals."

my_dog = Dog("Rover")

Properties & Methods

class Circle:
    def __init__(self, radius):
        self.radius = radius

    @property
    def area(self):
        return 3.14159 * self.radius ** 2

8. Strings & Formatting

Basic String Usage

# Literals
text = "Hello"
quote = 'Python'
multiline = """Line one\nLine two"""
raw_path = r"C:\Users\Name"

# Concatenation & repetition
print(text + " " + quote)
print("ha" * 3)

Formatting Methods

# f-strings (best for readability)
name = "Ava"
age = 28
print(f"{name} is {age} years old.")

# str.format()
print("{0} scored {1:.1f}%".format(name, 92.384))

# Percent formatting
print("%s has %d messages" % (name, 5))

Useful String Methods

raw = "  Python is fun!  "
print(raw.strip())                       # "Python is fun!"
print(raw.upper())                       # "  PYTHON IS FUN!  "
print(raw.lower())                       # "  python is fun!  "
print(raw.replace("fun", "easy")) # "  Python is easy!  "
print(raw.split())                       # ['Python', 'is', 'fun!']
print("-".join(["a", "b", "c"])) # "a-b-c"
print("Python".startswith("Py")) # True
print("Python".endswith("on"))   # True

9. Built-in Functions & Common Methods

Essential Built-ins

# Numeric helpers
print(abs(-7))         # 7
print(round(3.14159, 2)) # 3.14

# Iterable helpers
numbers = [1, 2, 3]
print(sum(numbers))               # 6
print(min(numbers), max(numbers)) # 1 3
print(len(numbers))               # 3

Iteration Helpers

# Index with enumerate
for index, item in enumerate(numbers, 1):
    print(index, item)

# Pair iterables with zip
names = ["Ada", "Leo"]
scores = [95, 88]
for name, score in zip(names, scores):
    print(name, score)

Functional Tools

# Transform values with map
values = [1, 2, 3]
doubled = list(map(lambda x: x * 2, values))

# Filter values
evens = list(filter(lambda x: x % 2 == 0, values))

# Check all / any conditions
print(all([1 < 2, 2 < 3]))  # True
print(any([0, 0, 4]))         # True
